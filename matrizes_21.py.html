import numpy

matriz_1 = numpy.zeros(shape=(2, 2))
matriz_2 = numpy.zeros(shape=(2, 2))
continuar = 'sim'


def somar(u, v):
    matriz_somada = numpy.zeros(shape=(2, 2))
    for p in range(len(v)):
        for c in range(len(u[0])):
            matriz_somada[p][c] = u[p][c] + v[p][c]
    return matriz_somada


def subtrair(u, v):
    matriz_subtraida = numpy.zeros(shape=(2, 2))
    for t in range(len(matriz_subtraida)):
        for k in range(len(matriz_subtraida[0])):
            matriz_subtraida[t][k] = u[t][k] - v[t][k]
    return matriz_subtraida


def multiplicar(u, v):
    matriz_multiplicada = numpy.matmul(u, v)
    return matriz_multiplicada


for i in range(len(matriz_1)):  # ler a matriz 1
    for j in range(len(matriz_1[0])):
        print('digite um numero para a matriz 1')
        matriz_1[i][j] = float(input())

for i in range(len(matriz_2)):  # ler a matriz 2
    for j in range(len(matriz_2[0])):
        print('digite um numero para a matriz 2')
        matriz_2[i][j] = float(input())


while continuar == 'sim':
    print('qual operaçao você deseja fazer? (somar, subtrair, multiplicar)')
    resposta_usuario = input()

    if resposta_usuario == 'somar':
        somando = somar(matriz_1, matriz_2)
        print(somando)
    elif resposta_usuario == 'subtrair':
        subtraindo = subtrair(matriz_1, matriz_2)
        print(subtraindo)
    elif resposta_usuario == 'multiplicar':
        multiplicando = multiplicar(matriz_1, matriz_2)
        print(multiplicando)

    print('deseja continuar?')
    continuar = input()

    if continuar == 'nao':
        break
    else:
        continuar = 'sim'
